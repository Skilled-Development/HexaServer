

use bytes::{Buf, BytesMut, BufMut};
use uuid::Uuid;

// Lee un VarInt desde el buffer y elimina el dato del buffer
pub fn read_varint(buf: &mut &[u8]) -> Option<i32> {
    let mut value = 0;
    let mut shift = 0;

    for i in 0.. {
        if i >= 5 { return None; } // Protege contra sobrelectura
        let byte = buf.get(0)?;
        *buf = &buf[1..]; // Avanza el buffer
        value |= ((byte & 0x7F) as i32) << shift;
        if byte & 0x80 == 0 {
            return Some(value);
        }
        shift += 7;
    }
    None
}

// Lee un Int (4 bytes) desde el buffer y elimina el dato del buffer
pub fn read_int(buf: &mut &[u8]) -> Option<i32> {
    if buf.len() < 4 {
        return None;
    }
    let int_bytes = &buf[..4];
    *buf = &buf[4..];
    Some(i32::from_be_bytes(int_bytes.try_into().unwrap()))
}

// Lee un String (VarInt longitud + bytes del string) desde el buffer y elimina el dato del buffer
pub fn read_string(buf: &mut &[u8]) -> Option<String> {
    let length = read_varint(buf)? as usize;
    if buf.len() < length {
        return None;
    }
    let string_bytes = &buf[..length];
    *buf = &buf[length..];
    Some(String::from_utf8_lossy(string_bytes).to_string())
}

// Lee un Long (8 bytes) desde el buffer y elimina el dato del buffer
pub fn read_long(buf: &mut &[u8]) -> Option<i64> {
    if buf.len() < 8 {
        return None;
    }
    let long_bytes = &buf[..8];
    *buf = &buf[8..];
    Some(i64::from_be_bytes(long_bytes.try_into().unwrap()))

}

pub fn write_varint(buf: &mut BytesMut, value: i32) {
    let mut value = value;
    while value & !0x7F != 0 {
        buf.put_u8((value & 0x7F | 0x80) as u8);
        value >>= 7;
    }
    buf.put_u8(value as u8);
}

// Escribe un entero de 4 bytes (Int) en el buffer
pub fn write_int(buf: &mut BytesMut, value: i32) {
    buf.put_i32(value);
}

// Escribe un String en el buffer precedido por su longitud (VarInt)
pub fn write_string(buf: &mut BytesMut, value: &str) {
    let length = value.len() as i32;
    write_varint(buf, length);
    buf.extend_from_slice(value.as_bytes());
}

// Escribe un entero de 8 bytes (Long) en el buffer
pub fn write_long(buf: &mut BytesMut, value: i64) {
    buf.put_i64(value);
}

// Escribe un byte (Byte) en el buffer
pub fn write_byte(buf: &mut BytesMut, value: u8) {
    buf.put_u8(value);
}

// Escribe un booleano (Boolean) en el buffer
pub fn write_boolean(buf: &mut BytesMut, value: bool) {
    buf.put_u8(if value { 1 } else { 0 });
}
pub fn read_unsigned_short(buf: &mut &[u8]) -> Option<u16> {
    if buf.len() < 2 {
        return None; // No hay suficiente data para leer un u16
    }
    // Lee los primeros 2 bytes y actualiza el buffer
    let (value_bytes, rest) = buf.split_at(2);
    *buf = rest; // Actualiza el buffer para eliminar los bytes leídos

    // Convierte los bytes a un u16
    let value = u16::from_be_bytes([value_bytes[0], value_bytes[1]]);
    Some(value)
}
